\begin{frame}{Defining the critical parameter $p_c$}
  Fix a lattice. Let
  \[
    \theta(p) \;=\; \P_p\!\big(\, 0 \leftrightarrow \infty \,\big)
  \]
  be the probability that the cluster of the origin is infinite.
  $\theta$ is monotone non-decreasing in $p$, and
  \[
    p_c \;:=\; \sup\{\, p \in [0,1] \,:\, \theta(p) = 0 \,\}
        \;=\; \inf\{\, p \in [0,1] \,:\, \theta(p) > 0 \,\}.
  \]

  \begin{itemize}
    \item $p < p_c$: with probability $1$, every cluster is finite
          (sub-critical phase).
    \item $p > p_c$: a unique infinite cluster exists almost surely
          (super-critical phase).
    \item $p = p_c$: \textbf{critical phase}. No infinite cluster, but
          clusters of every scale --- the scale-invariant regime where
          $\E V(r) \sim r^{d_f}$ lives.
  \end{itemize}

  \vspace{0.4em}
  \begin{center}
    \begin{tabular}{lccc}
      \toprule
      lattice              & honeycomb            & square                       & triangular \\
      vertex degree        & $3$                  & $4$                          & $6$        \\
      $p_c$                & $\approx 0.6962$     & $\approx 0.59274621$         & $= 0.5$    \\
      \bottomrule
    \end{tabular}
  \end{center}

  \TODO{Higher vertex degree means it is easier for the cluster to

        \textbf{exact} (Sykes--Essam duality); square is only known
        numerically. See~\cite{grimmett1999percolation}.}
\end{frame}

\begin{frame}{What physicists measure vs.\ what they want}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{block}{Want (unobservable)}
        \centering $\E V(r)$ \\[0.2em]
        \footnotesize the population mean
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Have (observable)}
        \centering $\overline V(r) \;=\; \dfrac{1}{n}\sum_{t=1}^{n} V_t(r)$ \\[0.2em]
        \footnotesize sample mean over $n$ independent trials
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.8em}
  \textit{Two reasons our log--log reading can be off:}
  \begin{itemize}
    \item[(a)] $\overline V(r) \ne \E V(r)$ --- random fluctuations,
               source of \textbf{variance};
    \item[(b)] $\E V(r)$ is only \textbf{asymptotically} a power law
               --- deterministic offset at small $r$, source of
               \textbf{bias}.
    \end{itemize}

  \vspace{0.4em}
  The next two frames formalise each source.
\end{frame}

\begin{frame}{Setup: a generic observable $Y_i$}
  Let $Y_i$ be a generic observable with $\gamma$ as the related
  critical exponent.

  \vspace{0.4em}
  \textbf{(A0)} $Y_i > 0$ \quad (so $\log Y_i$ is defined).

  \vspace{0.3em}
  \textbf{(A1) Assumption 1.}
  \[
    \E Y_i \;=\; a_0\, i^{\gamma}\,
       \exp\!\Big(\, a_1\, i^{-1}
                  \;+\; \phi(i)\, i^{-2} \,\Big),
    \qquad \phi \text{ bounded.}
  \]

  \vspace{0.4em}
  Identification for our running example:
  $Y_i \leftrightarrow V(r)$ with $i = r$, $\gamma \leftrightarrow d_f$.

  \vspace{0.4em}
  {\footnotesize\textbf{Reading the expansion.}
  \begin{itemize}
    \item $a_0\, i^{\gamma}$ --- the power law we are after.
    \item $a_1 / i$ --- leading finite-size correction (boundary,
          discreteness). Its log is the source of
          \textbf{deterministic bias} at small $i$.
    \item $\phi(i) / i^2$ --- bounded remainder, one power faster.
    \item This is \textbf{not} exponential decay: the exponential is
          just a tidy way to write the multiplicative correction
          $a_0\, i^{\gamma}(1 + a_1/i + \cdots)$ in a form whose log
          is additive.
  \end{itemize}}
\end{frame}

\begin{frame}{Chalkboard: per-scale decomposition}
  Define the random ratio
  $\;\overline{\xi}_k \;:=\; \overline Y_{\rho^k}\,/\,\E Y_{\rho^k}\;$
  (mean $1$).

  \vspace{0.2em}
  We can decompose our log--log plot as follows: write
  $\log \overline Y_{\rho^k} = \log \E Y_{\rho^k} + \log \overline{\xi}_k$
  and substitute $i = \rho^k$ in the expansion of the previous slide.
  \[
    \boxed{\;
      \log \overline Y_{\rho^k} \;=\;
        \underbrace{\log a_0 + k\,\gamma \log \rho}_{\text{signal (linear in }k\text{)}}
        \;+\;
        \underbrace{a_1\, \rho^{-k} + \phi(\rho^k)\, \rho^{-2k}}_{\text{finite-size bias (deterministic)}}
        \;+\;
        \underbrace{\log \overline{\xi}_k}_{\text{random}}\;.\;}
  \]

  \vspace{0.2em}
  {\footnotesize\textbf{Notation.} Write $\alpha = \log a_0$ and
  $\beta = \gamma \log \rho$ for the signal coefficients.}

  \vspace{0.4em}
  \begin{itemize}
    \item Signal: linear in $k$ with slope $\beta = \gamma \log \rho$.
    \item Bias: decays \textbf{geometrically} in $k$ (since $\rho > 1$).
          After a few scales it sits below statistical noise.
    \item Random part: $\E \log \overline{\xi}_k \approx 0$,
          $\Var \log \overline{\xi}_k \asymp \sigma_k^2 / n$.
  \end{itemize}

\begin{frame}{Bias--variance trade-off}
  Every term in the per-scale decomposition is now formal:
  \begin{itemize}
    \item the finite-size bias has size $\asymp \rho^{-k}$
          (deterministic, shrinks with $k$);
    \item the random part has size $\asymp \sigma_k/\sqrt{n}$
          (stochastic, controlled by simulation cost).
  \end{itemize}

  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Small $r$  (cheap)}
        $\checkmark$ many trials affordable in a fixed compute budget
        --- \textbf{low variance} of $\overline V(r)$.\\[0.4em]
        $\times$ finite-size correction $a_1 / r$ is large ---
        \textbf{high bias}.
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Large $r$  (expensive)}
        $\checkmark$ asymptotic regime: $a_1 / r \to 0$ ---
        \textbf{low bias}.\\[0.4em]
        $\times$ each trial costs more time, fewer trials fit in the
        budget --- \textbf{high variance}.
      \end{exampleblock}
    \end{column}
  \end{columns}

    \textbf{statistical decision}, not an aesthetic one.

\section{The log--log estimator and its three-part split}
% target: ≈ 6 min
% =====================================================================

\begin{frame}{Linear regression}
  Data: $\big(k,\, \log \overline Y_{\rho^k}\big)$ for
  $k = m_0 + 1, \ldots, m_0 + m$.

  \vspace{0.4em}
  The \textbf{ordinary least squares} (OLS) slope has the closed form
  \[
    \hat\beta \;=\; \sum_{k\, =\, m_0 + 1}^{m_0 + m}
        w_{k,m}\, \log \overline Y_{\rho^k},
    \qquad
    w_{k,m} \;=\; \dfrac{12\,\big(k - m_0 - (m+1)/2\big)}{m\,(m^2 - 1)}.
  \]

  Hence the estimator of the target exponent:
  \[
    \hat\gamma \;=\; \dfrac{\hat\beta}{\log \rho}.
  \]

  \TODO{The weights $w_{k,m}$ are exactly the closed-form OLS
        coefficients for fitting a line to equally-spaced abscissas
        $k$. They are linear in $k$, symmetric about the midpoint
        $m_0 + (m+1)/2$, and have two key identities (next slide).
        Two free knobs in the estimator:
        $m_0$ --- how many small-$r$ scales to drop (bias control),
        and $m$ --- window width (variance control). Both are tuned
        in §6.}
\end{frame}

\begin{frame}{Linear regression as a linear functional}
  Recall $\;\hat\beta \;=\; \displaystyle\sum_{k} w_{k,m}\,
        \log \overline Y_{\rho^k}\;$ --- a
  \textbf{deterministic linear combination} of the
  $\log \overline Y_{\rho^k}$'s.

  \vspace{0.4em}
  Two weight identities:
  \begin{itemize}
    \item $\displaystyle \sum_{k = m_0 + 1}^{m_0 + m} w_{k,m}
              \;=\; 0$
          \quad (kills any constant offset, e.g.\ $\log a_0$)
    \item $\displaystyle \sum_{k = m_0 + 1}^{m_0 + m} k\, w_{k,m}
              \;=\; 1$
          \quad (picks out the slope coefficient)
  \end{itemize}

  \vspace{0.4em}
  Useful consequence:
  \[
    \sum_{k = m_0 + 1}^{m_0 + m} w_{k,m}^{\,2}
    \;=\;
    \frac{12}{m(m^2-1)}
      \;\asymp\; \dfrac{1}{m^{3}}.
  \]

  \TODO{Linear functionals of random variables commute with the
        CLT, so once we know each summand obeys a CLT we get a CLT
        for $\hat\beta$ for free. The identity
        $\sum w_{k,m}^2 \asymp 1/m^3$ is where the $\sqrt{nm^3}$
        rate of the main theorem comes from:
        $\Var \hat\beta \approx \Var(\log \overline Y) \cdot
        \sum w^2 \asymp (1/n) \cdot (1/m^3)$, so the standard
        error is $\asymp 1/\sqrt{nm^3}$.}
\end{frame}

\begin{frame}{The three-part split}
  Apply $\;\sum_k w_{k,m}(\cdot)\;$ to the per-scale identity from §3:
  \[
    \hat\beta \;=\; \sum_k w_{k,m}\Bigl[
        \underbrace{\log a_0 + k\,\beta}_{\text{signal}}
        \;+\; \underbrace{a_1\rho^{-k} + \phi(\rho^k)\rho^{-2k}}_{\text{det.\ bias}}
        \;+\; \underbrace{\log\overline{\xi}_k}_{\text{random}}
      \Bigr].
  \]
  Weight identities (§4): $\sum w = 0$ kills $\log a_0$; $\sum k w = 1$
  picks out $\beta$. Hence $\hat\beta - \beta$ retains only the bias
  and random pieces.

  \vspace{0.3em}
  \textbf{Taylor-expand the random piece.} On the good event
  $\mathcal G_k := \{|\overline{\xi}_k - 1| \le 1/2\}$,
  $\;\log \overline{\xi}_k = (\overline{\xi}_k - 1) + q(\overline{\xi}_k - 1)\;$
  with $|q(x)| \le x^2$.

  \vspace{0.3em}
  \textbf{Collect:}
  \[
    \boxed{\;
      \hat\beta - \beta
        \;=\;
        \underbrace{\sum_k w_{k,m}\bigl(a_1\rho^{-k} + \phi(\rho^k)\rho^{-2k}\bigr)}_{\displaystyle R(m,m_0)}
        \;+\;
        \underbrace{\sum_k w_{k,m}(\overline{\xi}_k - 1)}_{\displaystyle S_{n,m}}
        \;+\;
        \underbrace{\sum_k w_{k,m}\,q(\overline{\xi}_k - 1)}_{\displaystyle Q_{n,m}}\;.\;}
  \]

\section{Main theorem: CLT for $\hat\gamma$}
% target: ≈ 10 min
% =====================================================================

\begin{frame}{Main theorem}
  Consider $\hat\gamma$ the log--log plot estimator for the critical
  exponent $\gamma$.

  \vspace{0.3em}
  \textbf{Assumptions on the model} (with $\xi_i := Y_i / \E Y_i$):
  \begin{itemize}
    \item[(A0)] $Y_i > 0$.
    \item[(A1)] $\E Y_i = a_0\, i^{\gamma}
                \exp\!\bigl(a_1\, i^{-1} + \phi(i)\, i^{-2}\bigr)$
                with $\phi$ bounded.
    \item[(A2)] $\E |\xi_i|^{2+\delta}$ and $\E |\log \xi_i|^{2+\delta}$
                are uniformly bounded in $i$ (some $\delta > 0$).
    \item[(A3)] $\sigma_k^2 := \Var(\log \xi_k)$ satisfies
                $\sigma_k \to \sigma_\infty \in (0,\infty)$.
  \end{itemize}

  \vspace{0.3em}
  \begin{block}{Theorem (Barbosa--Teixeira--Oliveira, 2026\textsuperscript{+})}
    Under (A0)--(A3), if $n, m, m_0 \to \infty$ along any sequence
    satisfying $m/n \to 0$ and $\rho^{-m_0} \ll \sqrt{m/n}$, then
    \[
      \sqrt{n\, m^{\,3}}\,\bigl(\hat\gamma - \gamma\bigr)
        \;\xrightarrow{\,d\,}\;
        \mathcal N\!\Bigl(\,0,\;
        \tfrac{12\,\sigma_\infty^{\,2}}{\log^{\,2}\!\rho}\,\Bigr).
    \]
  \end{block}
\end{frame}

\begin{frame}{Proof strategy}
  Recall the three-part split (§4):
  \[
    \hat\beta - \beta \;=\;
        \underbrace{R(m, m_0)}_{\text{deterministic bias}}
        \;+\;
        \underbrace{S_{n,m}}_{\text{linear stochastic}}
        \;+\;
        \underbrace{Q_{n,m}}_{\text{quadratic remainder}}\, .
  \]

  \vspace{0.4em}
  Multiply by $\sqrt{n\, m^{\,3}}$ and treat each piece:
  \begin{itemize}
    \item $\sqrt{n m^3}\, S_{n,m}\;\xrightarrow{\,d\,}\;\mathcal N(0,
          12\sigma_\infty^2)$ \quad (Lindeberg--Feller)
    \item $\sqrt{n m^3}\, Q_{n,m}\;\xrightarrow{\,p\,}\; 0$
          \quad (good event $+$ moment bound)
    \item $\sqrt{n m^3}\, R(m,m_0)\;\to\; 0$
          \quad (deterministic, controlled by $m_0$)
  \end{itemize}

  \vspace{0.4em}
  Divide by $\log\rho$ ($\hat\gamma = \hat\beta/\log\rho$): variance
  becomes $12\sigma_\infty^2/\log^2\rho$.
\end{frame}

\begin{frame}{$S_{n,m}$ via Lindeberg--Feller}
  $S_{n,m}$ is the OLS slope applied to the centred stochastic part
  (with $\overline{\xi}_k = \overline Y_{\rho^k}/\E Y_{\rho^k}$, $\E\overline{\xi}_k = 1$):
  \[
    S_{n,m} \;=\; \sum_{k = m_0+1}^{m_0+m} w_{k,m}\,
        \bigl(\overline{\xi}_k - 1\bigr).
  \]
  A triangular array of \textbf{weighted i.i.d.\ averages}:
  $n$ replicas inside each $\overline{\xi}_k$, $m$ weighted terms in $k$.

  \vspace{0.4em}
  \textbf{Variance.} Using independence across $k$ (the $Y_i$'s at
  different scales are computed on disjoint annuli):
  \[
    \Var(S_{n,m}) \;=\; \tfrac{1}{n}
        \sum_{k} w_{k,m}^{\,2}\, \Var(\xi_k)
        \;\asymp\; \tfrac{1}{n m^3}\, \sigma_\infty^2.
  \]

  \vspace{0.2em}
  \textbf{Lyapunov $(2+\delta)$-condition.} Under bounded
  $(2+\delta)$-th moments of $\xi_k$,
  we can show
  \[
    \frac{1}{s_{n,m}^{\,2+\delta}}\sum_{k=m_0+1}^{m}
          \E\!\left[\bigl|w_{k,m}(\bar{\xi}_{k}-1)\bigr|^{2+\delta}\right]
    \;\xrightarrow[m,n\to\infty]{}\; 0.
  \]
  Lindeberg--Feller gives the CLT.
\end{frame}

\begin{frame}{$Q_{n,m}$ Taylor remainder vanishes faster}
  From §4: on the \textbf{good event}
  $\mathcal{G}_k = \{|\overline{\xi}_k - 1| \le 1/2\}$,
  $\;q(\overline{\xi}_k - 1) = \log(\overline{\xi}_k) - (\overline{\xi}_k - 1)\;$
  satisfies $|q(\overline{\xi}_k-1)| \le (\overline{\xi}_k - 1)^{\,2}$. So
  \[
    |Q_{n,m}| \;\le\; \sum_k |w_{k,m}|\, (\overline{\xi}_k - 1)^{\,2}
                 \;+\; \text{bad-event term}.
  \]

  \vspace{0.4em}
  \textbf{Good event.} $\E(\overline{\xi}_k - 1)^{\,2} \asymp
  \sigma_k^2/n$, so $\E|Q_{n,m}^{\text{good}}| \lesssim
  (1/n)\sum_k |w_{k,m}| \asymp 1/(nm)$. Multiplying by $\sqrt{nm^3}$
  gives $\sqrt{m/n} \to 0$.

  \vspace{0.4em}
  \textbf{Bad event.} Use union bound and Chebyshev to
  bound $\P(\mathcal{G}_k^c) \le 4\sigma_k^2/n$. Then use
  Cauchy--Schwarz to bound $\E[Q_{n,m}\,\mathbbm{1}_{\mathcal{G}_k^c}]$.
\end{frame}

\begin{frame}{$R(m,m_0)$ finite-size bias --- the geometric scale wins}
  The $R(m,m_0)$ term is \textbf{deterministic} bias from the linear
  regression:
  \[
    R(m,m_0) \;=\; \sum_{k = m_0+1}^{m_0+m} w_{k,m}\,
        \Bigl(a_1\,\rho^{-k} + \phi(\rho^k)\,\rho^{-2k}\Bigr).
  \]

  \vspace{0.4em}
  Under reasonable assumptions both correction terms decay
  \textbf{geometrically} in $k$, and $\sum_k |w_{k,m}| \asymp 1/m$.
  The leading $\rho^{-k}$ term, summed against the symmetric weights,
  gives
  \[
    R(m,m_0) \;\asymp\; \frac{\rho^{-m_0}}{m^2}.
  \]

  \vspace{0.4em}
  \textbf{Condition for negligibility:}
  $\sqrt{nm^3}\cdot R(m,m_0) \to 0\;\Longleftrightarrow\;
  \rho^{-m_0} \ll \sqrt{m/n}$ (equivalently
  $\rho^{m_0} \gg \sqrt{n/m}$). This is the second growth condition
  in the theorem --- and the design knob $m_0$ that drives §6.
\end{frame}

\section{Lower bound and optimality}
% target: ≈ 4 min
% =====================================================================

\begin{frame}{What rate does the log--log plot achieve?}
  \textbf{Cost model.} A single trial at scale $L=\rho^{k}$ costs
  $\asymp L^{\,d}$ operations, where $d$ is the per-trial cost
  exponent of the simulator (e.g.\ $d=2$ for arena BFS in §6).
  With $n$ replicas, the \emph{simulation budget} is
  \[
    B \;=\; n\,\rho^{\,(m_0+m)\,d}.
  \]

  \vspace{0.3em}
  \textbf{A CLT-admissible schedule.} Heuristically (balance the
  two surviving error terms and saturate the budget):
  \[
    m_0(B) \;=\; \Bigl\lfloor\tfrac{1}{d+2}\,\log_\rho\! B \Bigr\rfloor,\quad
    m(B)  \;=\; \lceil \log_\rho\!\log B \rceil,\quad
    n(B)  \;=\; \bigl\lfloor B\,/\,\rho^{(m_0+m)\,d}\bigr\rfloor.
  \]
  One checks $m/n \to 0$ and $\rho^{-m_0}\!\ll\!\sqrt{m/n}$, so both
  hypotheses of the CLT (§5) hold.

  \vspace{0.3em}
  Plug into the two surviving terms from §5
  ($R\asymp\rho^{-m_0}/m^{2}$ and $\Var(\hat\gamma)\asymp 1/(n m^{3})$):
  \[
    \E\bigl[(\hat\gamma-\gamma)^{\,2}\bigr]
    \;\le\;
    \underbrace{\rho^{-2m_0}/m^{4}}_{\text{bias}^2}
    \;+\;
    \underbrace{C/(n m^{3})}_{\text{variance}}
    \;\le\;
    C'\,B^{-\,2/(d+2)}\,(\log\log B)^{c}.
  \]

  \vspace{0.3em}
  \textbf{Question.} Is the exponent $2/(d+2)$ \emph{sharp}, or could
  a cleverer estimator do polynomially better?
\end{frame}

\begin{frame}{Answer: $\boldsymbol{B^{-2/(d+2)}}$ is sharp (up to logs)}
  \begin{block}{Theorem (minimax lower bound)}
    Take $\mathcal{P}$ = distributions $\mathbb{W}=(\mathbb{W}_L)_{L\in\mathbb{N}}$
    with $\bigl|\E\mathbb{W}_L - a_0\, L^{\gamma}\bigr|\le \exp(a_1/L)$
    and $\|\mathbb{W}_L/\E\mathbb{W}_L\|_{2+\delta}<M$. Then for every
    $\log B \ge 2(d+2)/d$,
    \[
      \inf_{\hat\gamma}\sup_{\mathbb{W}\in\mathcal{P}}\;
        \E_{\mathbb{W}}\!\bigl[(\hat\gamma - \gamma)^{\,2}\bigr]
      \;\ge\;
      \frac{(d+2)^{2}}{8}\,\frac{c^{2}\,e^{-a^{2}/2}}{\log^{2}\!B}\;
          B^{-\,2/(d+2)}.
    \]
  \end{block}

  \vspace{0.4em}
  Combined with the previous slide: the log--log plot is
  \textbf{minimax-rate-optimal up to a $\log^{\,2}\!B$ factor} ---
  no estimator can do polynomially better.

  \vspace{0.4em}
  \footnotesize Closely related: Carpentier \& Kim
  \cite{carpentierKim2015tail} prove the analogous rate for the
  tail coefficient of \emph{second-order Pareto} distributions, with
  essentially the same two-point KL construction.
\end{frame}

\begin{frame}{Proof idea --- two-point reduction}
  Pick \emph{two} candidate models $\mathbb{X},\mathbb{Y}\in\mathcal{P}$
  whose parameters $\gamma$ differ by $s$. The minimax risk
  is lower-bounded by separating these two:
  \begin{block}{Two-point lemma (adapted from \cite{minimax_lowerBoundsJennifer})}
    \[
      \inf_{\hat\gamma}\sup_{\mathbb{W}\in\mathcal{P}}
        \E_{\mathbb{W}}\bigl[(\hat\gamma - \gamma)^{\,2}\bigr]
      \;\ge\; \tfrac{1}{4}\,s^{2}\,\exp\bigl(-\mathrm{KL}(\mathbb{X}\,\|\,\mathbb{Y})\bigr).
    \]
  \end{block}

  \vspace{0.4em}
  \textbf{Our construction.} For each scale $L=\rho^{k}$,
  \[
    X_L \;\sim\; e^{Z},\qquad
    Y_L \;\sim\;
      \begin{cases} e^{Z} & L < L^{\ast},\\[0.2em]
                    e^{Z + \lambda_L} & L \ge L^{\ast}, \end{cases}
        \qquad Z\sim\mathcal{N}(0,1).
  \]
  \quad with $L^{\ast}=B^{1/(d+2)}$,
  $\lambda_L = \gamma \log L$,
  $\gamma = \log\!\bigl(1-1/L^{\ast}\bigr)\big/\log(1/L^{\ast})$.

  \vspace{0.4em}
  Direct check: $\mathbb{Y}$ stays inside $\mathcal{P}$ (the leading
  expansion holds at order $L^{-1}$), $\E X_L = 1$, and
  $\E Y_L = L^{\gamma}\!\cdot\!\mathbbm{1}_{\{L\ge L^{\ast}\}}+\mathbbm{1}_{\{L<L^{\ast}\}}$.
\end{frame}

\begin{frame}{Proof idea --- the ``enemy'' picks the worst budget split}
  The KL between the two product measures factorises over scales:
  \[
    \mathrm{KL}(\mathbb{X}\,\|\,\mathbb{Y})
      \;=\; \tfrac{1}{2}\sum_{L\ge L^{\ast}} n_L(B)\,\gamma^{\,2}\log^{2}\!L,
  \]
  where $n_L(B)$ is the number of trials the statistician spends at
  scale $L$, subject to the budget $\sum_L n_L\,L^{\,d}\le B$.

  \vspace{0.4em}
  \textbf{The enemy} (Nature) maximises the KL. Change of variable
  $M_L = n_L L^{\,d}/B$:
  \[
    \max_{(M_L)}\;\sum_{L\ge L^{\ast}} B\,M_L\,L^{-d}\log^{2}\!L
        \quad\text{s.t.}\quad \sum_L M_L \le 1.
  \]
  Linear in $M_L$ on the simplex $\Rightarrow$ optimum at an extreme point:
  put all mass on the $L$ that maximises $L^{-d}\log^{2}\!L$. Calculus shows
  this is $L^{\ast}$ as soon as $\log B\ge 2(d+2)/d$.

  \vspace{0.4em}
  Substituting $L^{\ast}=B^{1/(d+2)}$ and bounding
  $\log^{2}(1-x)\le a^{2}x^{2}$ gives $\mathrm{KL}\le a^{2}/2$, a
  \emph{constant}. The $s^{2}\!\cdot\! e^{-\mathrm{KL}}$ in the two-point
  lemma then yields the announced rate.
\end{frame}

\section{Wrap-up}
% target: ≈ 2 min
% =====================================================================

\begin{frame}{Take-aways}
  \begin{enumerate}
    \item \textbf{The physicists' log--log plot has a rigorous
          statistical theory.} A CLT for $\hat\gamma$ with explicit
          rate $\sqrt{n m^{\,3}}$ and limiting variance
          $12\sigma_\infty^{\,2}/\log^{\,2}\!\rho$ --- assumptions
          (A0)--(A3) on the model, joint scaling
          $\rho^{-m_0}\!\ll\!\sqrt{m/n}$ on the regression design.
    \item \textbf{Three explicit error terms, three different rates.}
          Deterministic bias $R$ ($\rho^{-m_0}/m^{2}$), linear stochastic $S$
          (Gaussian at $1/\sqrt{n m^{\,3}}$), Taylor remainder $Q$
          ($1/(nm)$). They each have their own knob --- $m_0$
          balances them.
    \item \textbf{$m_0$ is the practical lever.} Empirically (§6),
          $m_0 \asymp \tfrac{1}{2}\log_\rho(n m^{\,3})$ gives the
          best bias--variance trade-off; $m_0 = \lfloor m/2\rfloor$
          is a robust default; $m_0 = 1$ leaves the leading bias
          on the table.
    \item \textbf{Near-minimax.} On the same budget $B = n\,\rho^{(m_0+m)d}$
          the achieved RMSE rate $B^{-1/(d+2)}$ matches the minimax
          lower bound up to a $\log B$ factor: no estimator can beat
          the log--log plot polynomially in the budget $B$.
  \end{enumerate}

  \vspace{0.4em}
  \textbf{Practical recipe.} Pick $m\asymp\log_\rho B$, $m_0$ as above,
  spend the remaining budget on replicas $n$; read off
  $\hat\gamma$ and its Wilson interval.
\end{frame}
