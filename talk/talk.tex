\begin{frame}{Defining the critical parameter $p_c$}
  Fix a lattice. Let
  \[
    \theta(p) \;=\; \P_p\!\big(\, 0 \leftrightarrow \infty \,\big)
  \]
  be the probability that the cluster of the origin is infinite.
  $\theta$ is monotone non-decreasing in $p$, and
  \[
    p_c \;:=\; \sup\{\, p \in [0,1] \,:\, \theta(p) = 0 \,\}
        \;=\; \inf\{\, p \in [0,1] \,:\, \theta(p) > 0 \,\}.
  \]

  \begin{itemize}
    \item $p < p_c$: with probability $1$, every cluster is finite
          (sub-critical phase).
    \item $p > p_c$: a unique infinite cluster exists almost surely
          (super-critical phase).
    \item $p = p_c$: \textbf{critical phase}. No infinite cluster, but
          clusters of every scale --- the scale-invariant regime where
          $\E V(r) \sim r^{d_f}$ lives.
  \end{itemize}

  \vspace{0.4em}
  \begin{center}
    \begin{tabular}{lccc}
      \toprule
      lattice              & honeycomb            & square                       & triangular \\
      vertex degree        & $3$                  & $4$                          & $6$        \\
      $p_c$                & $\approx 0.6962$     & $\approx 0.59274621$         & $= 0.5$    \\
      \bottomrule
    \end{tabular}
  \end{center}

  \TODO{Higher vertex degree means it is easier for the cluster to
        grow, and so $p_c$ is lower. Triangular $p_c = 1/2$ is
        \textbf{exact} (Sykes--Essam duality); square is only known
        numerically. See~\cite{grimmett1999percolation}.}
\end{frame}

\begin{frame}{What physicists measure vs.\ what they want}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{block}{Want (unobservable)}
        \centering $\E V(r)$ \\[0.2em]
        \footnotesize the population mean
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Have (observable)}
        \centering $\overline V(r) \;=\; \dfrac{1}{n}\sum_{t=1}^{n} V_t(r)$ \\[0.2em]
        \footnotesize sample mean over $n$ independent trials
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.6em}
  \textbf{Working hypothesis.} The population mean admits the
  asymptotic expansion
  \[
    \E Y_i \;=\; a_0\, i^{\gamma}\,
       \exp\!\Big(\, a_1\, i^{-1}
                  \;+\; \phi(i)\, i^{-2} \,\Big),
    \qquad \phi \text{ bounded.}
  \]
  {\footnotesize\textbf{Remark.} This is \textbf{not} exponential decay.
  The exponential is just a convenient way to write the multiplicative
  correction
  $\;\E Y_i \approx a_0\, i^{\gamma} + a_1\, i^{\gamma - 1} + \cdots\;$
  in a form whose $\log$ is additive.}

  \vspace{0.4em}
  Identifications for our running example:
  $Y_i \leftrightarrow V(r)$ with $i = r$, $\gamma \leftrightarrow d_f$.

  \TODO{$a_0$ is a lattice-dependent amplitude (the intercept on the
        log--log panel). $a_1$ is the leading finite-size correction
        coefficient --- encodes boundary / discreteness effects.
        $\phi$ is the remainder, assumed only \textbf{bounded}.
        Higher-order corrections sit inside $\phi$ and decay one power
        of $i$ faster than $a_1/i$.}
\end{frame}

\begin{frame}{Bias--variance trade-off}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Small $r$  (cheap)}
        $\checkmark$ many trials affordable in a fixed compute budget
        --- \textbf{low variance} of $\overline V(r)$.\\[0.4em]
        $\times$ finite-size correction $a_1 / r$ is large ---
        \textbf{high bias}.
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Large $r$  (expensive)}
        $\checkmark$ asymptotic regime: $a_1 / r \to 0$ ---
        \textbf{low bias}.\\[0.4em]
        $\times$ each trial costs more time, fewer trials fit in the
        budget --- \textbf{high variance}.
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.8em}
  \begin{center}
    \large
    Choosing which scales to use should be a
    statistical decision, not an aesthetic one.
  \end{center}

\begin{frame}{Chalkboard: log--log linearisation}
  Take $\log$ of the expansion above and substitute $i = \rho^k$:
  \[
    \log \E Y_{\rho^k} \;=\;
      \underbrace{\log a_0 \;+\; k\, \gamma \log \rho}_{\text{signal (linear in }k\text{)}}
      \;+\;
      \underbrace{a_1\, \rho^{-k} \;+\; \phi(\rho^k)\, \rho^{-2k}}_{\text{finite-size correction (geometric in }k\text{)}}.
  \]

  \vspace{0.4em}
  \begin{itemize}
    \item In the variable $k$, the leading term is linear with slope
          $\gamma \log \rho$.
    \item The correction decays \textbf{geometrically} in $k$ (since
          $\rho > 1$): after a few scales it is statistical noise, not
          signal.
    \item Therefore the OLS slope of $\log \overline Y_{\rho^k}$ on
          $k$ is a natural estimator of $\gamma \log \rho$, hence of
          $\gamma$.
  \end{itemize}

  \TODO{Board steps:
        (1) Start from $\E Y_i = a_0\, i^{\gamma}\, \exp(a_1/i + \phi(i)/i^2)$.
        (2) Take log: $\log \E Y_i = \log a_0 + \gamma \log i + a_1/i + \phi(i)/i^2$.
        (3) Substitute $i = \rho^k$: $\log i = k \log \rho$, $1/i = \rho^{-k}$.
        (4) Box the result. Highlight that the geometric grid
        $\rho^k$ is what makes the correction decay
        \textbf{exponentially fast} in $k$ --- the whole reason for
        using a geometric scale schedule.}
\end{frame}
